Usage

Browser

In any web page:

<script src="http://web-data-connector-wrapper.com/wdcw.2.0.0.standalone.min.js)"></script>
<script>
  var wdcw = require('web-data-connector-wrapper');
  var example = new wdcw();

  example.hello();
</script>

Node.js

var wdcw = require('./src/web-data-connector-wrapper.js');
var example = new wdcw();

example.hello();
